    let progresoEquilibrio = 50;

    // Función para renderizar dinámicamente la información
    function mostrarPueblo(puebloKey, elementoCard) {
        // Remover clase activa previa
        document.querySelectorAll('.pueblo-card').forEach(card => card.classList.remove('active'));
        // Agregar clase activa a la tarjeta seleccionada
        elementoCard.classList.add('active');

        const pueblo = datosPueblos[puebloKey];
        const contenedor = document.getElementById('contenedor-info');

        contenedor.innerHTML = `
            <div class="info-header">
                <h2>${pueblo.nombre}</h2>
            </div>
            <div class="info-content">
                <div class="info-block">
                    <strong>🍂 Costumbres y Tradiciones:</strong>
                    <p>${pueblo.costumbres}</p>
                </div>
                <div class="info-block">
                    <strong>🧥 Vestimenta e Identidad:</strong>
                    <p>${pueblo.vestimenta}</p>
                </div>
                <div class="info-block">
                    <strong>🏔️ Rol Espiritual y Territorial:</strong>
                    <p>${pueblo.rol}</p>
                </div>
                <div class="info-block">
                    <strong>💡 Dato Destacado:</strong>
                    <p>${pueblo.dato}</p>
                </div>
            </div>
        `;

        // Aumentar sutilmente la barra de equilibrio por leer e investigar
        modificarEquilibrio(5);
    }

    // Lógica interactiva para la barra de equilibrio
    function modificarEquilibrio(cantidad) {
        progresoEquilibrio = Math.min(progresoEquilibrio + cantidad, 100);
        document.getElementById('balance-bar').style.width = progresoEquilibrio + '%';
        
        let estadoText = "Equilibrio Estable";
        if(progresoEquilibrio >= 85) estadoText = "Armonía Ancestral Plena ✨";
        else if(progresoEquilibrio >= 65) estadoText = "Territorio Sanando 🌱";

        document.getElementById('balance-text').innerText = `🌱 Equilibrio Espiritual de la Sierra: ${progresoEquilibrio}% (${estadoText})`;
    }

    // Registrar pensamientos en el simulador
    function registrarOfrenda() {
        const input = document.getElementById('input-pensamiento');
        const texto = input.value.trim();

        if (texto === "") {
            alert("Por favor, escribe un pensamiento respetuoso antes de enviarlo a la Sierra.");
            return;
        }

        const log = document.getElementById('log-pensamientos');
        const nuevoItem = document.createElement('div');
        nuevoItem.className = 'thought-item';
        nuevoItem.innerText = texto;
        
        // Insertar al inicio de la bitácora
        log.insertBefore(nuevoItem, log.firstChild);
        
        // Limpiar input y subir el medidor de equilibrio
        input.value = "";
        modificarEquilibrio(10);
    }

    // Inicializar con el pueblo Kogui al cargar la página
    window.onload = function() {
        mostrarPueblo('kogui', document.querySelector('.pueblo-card'));
        document.getElementById('balance-bar').style.width = '50%';
    };
